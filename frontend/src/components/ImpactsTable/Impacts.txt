import React from "react";

const ImpactsTable = ({ impact }) => {
    return (
        <div className="impact-card-container">
        <h4 className="impact-card-title">Liste des impacts</h4>
        <div className="impact-card-line"></div>

        {impact ? (
          <div className="impact-card">
            <div className="impact-header">
              <span className="impact-meta">
                <strong>Thématique :</strong> {impact.thematiqueNom}
              </span>
              <span className="impact-meta">
                <strong>Métier :</strong> {impact.metierNom}
              </span>
              <span className="impact-meta">
                <strong>Composante :</strong> {impact.composanteNom}
              </span>
            </div>
            <div className="impact-descriptions">
              {Array.isArray(impact.descriptions) &&
                impact.descriptions.map((desc, index) => (
                  <div key={index} className="impact-item">
                    <span className="impact-icon">✔️</span>
                    <p>{desc}</p>
                  </div>
                ))}
            </div>
          </div>
        ) : (
          <p className="text-muted">
            Aucun impact sélectionné. Veuillez choisir une thématique, un métier et une composante.
          </p>
        )}
      </div>
      
    );
};

export default ImpactsTable;
